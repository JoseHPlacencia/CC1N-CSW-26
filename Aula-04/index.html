def integral_trapezio(f, a, b, n=1000):
    h = (b - a) / n
    soma = 0.5 * (f(a) + f(b))
    
    for i in range(1, n):
        soma += f(a + i * h)
    
    return soma * h

# Exemplo: ∫ de 0 a 2 de x²
f = lambda x: x**2

resultado = integral_trapezio(f, 0, 2)
print("Resultado aproximado:", resultado)
